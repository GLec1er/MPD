‚úÖ –ü—Ä–∏–º–µ—Ä:
```python

        self.name = name

    def __hash__(self):
        return hash((self.name, self.age))

p1 = Person("Alice", 30)
p2 = Person("Alice", 30)

s = {p1, p2}  # –û–ö: p1 –∏ p2 —Å—á–∏—Ç–∞—é—Ç—Å—è —Ä–∞–≤–Ω—ã–º–∏ –∏ –Ω–µ –¥—É–±–ª–∏—Ä—É—é—Ç—Å—è
```
